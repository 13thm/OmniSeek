# docker es 构建
```bash
mkdir -p D:\environment\docker-save\es\data D:\environment\docker-save\es\plugins D:\environment\docker-save\es\config

# 启动 ES 容器，暴露 9200/9300 端口，挂载本地目录
docker run -d ^
  --name es717 ^
  --net es-network ^
  -p 9200:9200 ^
  -p 9300:9300 ^
  -e "discovery.type=single-node" ^
  -e "ES_JAVA_OPTS=-Xms1g -Xmx1g" ^
  -v D:\environment\docker-save\es\data:/usr/share/elasticsearch/data ^
  -v D:\environment\docker-save\es\plugins:/usr/share/elasticsearch/plugins ^
  -v D:\environment\docker-save\es\config:/usr/share/elasticsearch/config ^
  elasticsearch:7.17.0
```
